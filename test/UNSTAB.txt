AUTO
MODE 4
INPUT "CPU NAME ";CPU$
INPUT "DATE CODE ";DATE$
INPUT "CLOCK SPEED ";SPEED$
MODE 4
VDU 23,1,0;0;0;0;
FOR F=0 TO 1
  FOR INSTR=0 TO 3
    FOR CF=0 TO 1
      code%=&E000
      screen%=&5800
      screen_size%=&2800
      line%=&140
      OSWORD=&FFF1
      FOR I%=0 TO 2 STEP 2
        P%=code%
        [OPT I%
        .test
        \ Clear the screen
        LD HL,screen%
        LD DE,screen%+1
        LD (HL),&00
        LD BC,screen_size%
        LDIR
        \ HL is a pointer into the screen, start at bottom line
        LD   HL, screen%+screen_size%-line%+7
        \ SET Y/X FLAGS
        LD DE,&0000
        .loop
        PUSH DE
        PUSH DE
        POP AF
        NOP
        NOP
        POP AF
        ]
        IF INSTR=1 THEN [OPT I%:NOP:]
        IF INSTR=2 THEN [OPT I%:LD R,A:]
        IF INSTR=3 THEN [OPT I%:XOR &00:]
        IF CF=0 THEN [OPT I%:CCF:]
        IF CF=1 THEN [OPT I%:SCF:]
        [OPT I%
        \ SAVE FLAGS
        PUSH AF
        POP BC
        LD A,C
        ]
        IF F=1 THEN [OPT I%:RRCA:RRCA:]
        [OPT I%
        AND &08
        ADD A,&F8  \ result in C
        RL (HL)    \ shift into screen byte
        INC D
        LD A,D
        AND &07
        JR NZ,loop
        LD BC,&0008
        ADD HL,BC
        DEC D
        INC D
        JR NZ,loop
        LD BC,&10000-&100-1
        ADD HL,BC
        INC E
        LD A,E
        AND &07
        JR NZ,loop
        LD BC,&10000-line%+8
        ADD HL,BC
        DEC E
        INC E
        JR NZ,loop
        LD A,&FF
        LD HL,block
        CALL OSWORD
        RET
        .block
        DEFB &0D
        DEFB &01
        DEFW screen%
        DEFW 0
        DEFW screen%
        DEFW 0
        DEFW screen_size%
        DEFB 0
        ]
      NEXT
      REPEAT
        CALL test
      UNTIL INKEY(-1)=0
      MOVE 1100,250
      DRAW 1100,100
      DRAW 1250,100
      MOVE 1100,250:DRAW 1080,230
      MOVE 1100,250:DRAW 1120,230
      MOVE 1250,100:DRAW 1230,120
      MOVE 1250,100:DRAW 1230,80
      VDU 31,33,26,ASC"F"
      VDU 31,36,29,ASC"A"
      VDU 28,32,31,39,0,30
      PRINT "CPU:"'CPU$
      VDU 31,0,3
      PRINT "DATE:"'DATE$'
      PRINT "CLOCK:"'SPEED$'
      PRINT "TESTING:";
      IF INSTR=0 PRINT "POP AF"
      IF INSTR=1 PRINT "POP AF"'"NOP"
      IF INSTR=2 PRINT "POP AF"'"LD R,A"
      IF INSTR=3 PRINT "POP AF"'"XOR &00"
      IF CF=0 PRINT "CCF" ELSE PRINT "SCF"
      PRINT "PUSH AF"'
      PRINT "GRAPH:"'CHR$(ASC("X")+F);" FLAG"'
      PRINT "X AXIS:"'"A:00-FF"'
      PRINT "Y AXIS:"'"F:00-FF"'
      REM TAKE A SCREEN SHOT
      *MOTOR 1
      T=TIME
      REPEAT UNTIL TIME>T+20
      *MOTOR 0
      TIME=T
      REPEAT UNTIL TIME>T+100
    NEXT
  NEXT
NEXT
